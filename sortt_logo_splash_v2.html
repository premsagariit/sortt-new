// SplashScreen.tsx — React Native (Expo) implementation hint
// v2: Assembly sequence with scrap items + drive exit

import { Animated, Easing } from 'react-native'

// ── Phase 1: Assembly (staggered part entries) ──────────────
// Each part uses a single Animated.Value starting at off-screen
// translateX/Y and springing to 0. Parts: cab, cargo, chassis, wheels.

cabX.setValue(-300)   // cab from left
cargoX.setValue(300)   // cargo from right
chassisY.setValue(120) // chassis from below

Animated.stagger(120, [
  Animated.spring(cabX,    { toValue: 0, damping: 12, stiffness: 180, useNativeDriver: true }),
  Animated.spring(cargoX,  { toValue: 0, damping: 12, stiffness: 180, useNativeDriver: true }),
  Animated.spring(chassisY,{ toValue: 0, damping: 14, stiffness: 200, useNativeDriver: true }),
  Animated.spring(wFrontXY, { toValue: 0, damping: 9,  stiffness: 220, useNativeDriver: true }),
  Animated.spring(wRearXY,  { toValue: 0, damping: 9,  stiffness: 220, useNativeDriver: true }),
]).start()

// ── Phase 2: Scrap items fall (staggered, ~220ms apart) ─────
const scrapItems = [scrapNewspaper, scrapRods, scrapFridge]
scrapItems.forEach((item, i) =>
  setTimeout(() =>
    Animated.spring(item, {
      toValue: 0, delay: 0,
      damping: 7, stiffness: 130,
      useNativeDriver: true
    }).start(),
  1220 + i * 220)
)

// ── Phase 3: Drive left at t=2.0s ───────────────────────────
setTimeout(() => {
  Animated.timing(truckGroupX, {
    toValue: -400, duration: 1000,
    easing: Easing.in(Easing.cubic),
    useNativeDriver: true
  }).start()
  // Start wheel spin loop simultaneously
  wheelSpinLoop.start()
}, 2000)

Implementation note: Scrap items use transform-box: fill-box with negative translateY so they start above the SVG viewport. The phone screen's overflow: hidden clips them until they fall into frame — no additional masking needed. Wheel spinning uses transform-box: fill-box; transform-origin: 50% 50% per wheel group for correct rotation about each wheel's own centre. Total splash completes within 7s. Auth check and rate prefetch should resolve within this window.
